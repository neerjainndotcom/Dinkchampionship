🎾 Connect Google Sheets

One-time setup — takes about 3 minutes. After this, everyone with the link sees live scores.

1
Open your Google Sheet → click Extensions → Apps Script
2
Delete everything in the editor, then paste this code:
const SHEET_NAME = "Matches"; function doGet(e) { const sheet = getOrCreateSheet(); const data = sheet.getDataRange().getValues(); const matches = []; for (let i = 1; i < data.length; i++) { matches.push({ a: data[i][0], b: data[i][1], sa: data[i][2] === "" ? null : data[i][2], sb: data[i][3] === "" ? null : data[i][3] }); } return ContentService.createTextOutput(JSON.stringify({ matches })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const body = JSON.parse(e.postData.contents); const sheet = getOrCreateSheet(); if (body.action === "reset") { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { sheet.getRange(i+1, 3, 1, 2).setValues([["", ""]]); } } else if (body.action === "set") { const row = body.idx + 2; sheet.getRange(row, 3, 1, 2).setValues([[ body.sa === null ? "" : body.sa, body.sb === null ? "" : body.sb ]]); } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["TeamA","TeamB","ScoreA","ScoreB"]); const matches = []; for (let i=0;i<8;i++) for (let j=i+1;j<8;j++) matches.push([i,j,"",""]); sheet.getRange(2,1,matches.length,4).setValues(matches); } return sheet; }
